二叉树遍历

Sep 22, 2023
2 minute read

二叉树遍历方式有三种

  • 前序遍历 先遍历根节点，再遍历左子树，再遍历右子树

  • 中序遍历 先遍历左子树，再遍历根节点，最后遍历右子树

  • 后序遍历 先遍历左子树，再遍历右子树，再遍历根节点

其实不难发现，遍历方式是根据根节点遍历的顺序来定义的

package main 

type TreeNode struct {
  Left *TreeNode
  Right *TreeNode
  Val int
}

func main(){
  root := &TreeNode{
    Left: &TreeNode{
      Left: &TreeNode{Val: 4}, 
      Right: &TreeNode{Val: 5}, 
      Val: 2,
    }, 
    Right: &TreeNode{
      Val: 3, 
      Left: &TreeNode{Val: 6},
      Right: &TreeNode{Val: 7, },
    }, 
    Val: 1,
  }

  preorder(root)
}

// 前序遍历
func preorder(root *TreeNode){
  if root == nil {
    return
  }

  fmt.Println("vav:", root.Val)
  preorder(root.Left)
  preorder(root.Right)
}

// 中序遍历
func inorder (root *TreeNode) {
  if root == nil {
    return 
  }

  inorder(root.Left)
  fmt.Println("inorder:", root.Val)
  inorder((root.Right))
}

// 后序遍历
func postorder(root *TreeNode) {
  if root == nil {
    return
  }

  postorder(root.Left)
  postorder(root.Right)
  fmt.Println("postorder:", root.Val)
}
